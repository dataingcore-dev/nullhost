المشروع 5: AI Document Chat

نظام RAG للإجابة على الأسئلة من المستندات باستخدام LLM + Vector Search

← العودة إلى الرئيسية

مشروع RAG (Retrieval-Augmented Generation) كامل: ارفع مستندات PDF، حللها إلى chunks، خزن embeddings في ChromaDB، واسأل أسئلة عبر واجهة FastAPI.

هيكل المشروع

project/ ├── app/ │ ├── __init__.py │ ├── main.py │ ├── ingestion.py # معالجة المستندات │ ├── vector_store.py # ChromaDB embeddings │ ├── rag.py # RAG pipeline │ └── routers/ │ ├── __init__.py │ ├── documents.py │ └── chat.py ├── requirements.txt └── .env

1. Ingestion — معالجة المستندات

# ingestion.py from langchain_community.document_loaders import PyPDFLoader from langchain.text_splitter import RecursiveCharacterTextSplitter def load_and_chunk(pdf_path: str) -> list: loader = PyPDFLoader(pdf_path) documents = loader.load() splitter = RecursiveCharacterTextSplitter( chunk_size=500, chunk_overlap=50, separators=["\n\n", "\n", ".", " "] ) chunks = splitter.split_documents(documents) return chunks

2. Vector Store — التخزين المتجه

# vector_store.py from langchain_community.embeddings import HuggingFaceEmbeddings from langchain_community.vectorstores import Chroma embeddings = HuggingFaceEmbeddings( model_name="sentence-transformers/all-MiniLM-L6-v2" ) vector_store = Chroma( collection_name="documents", embedding_function=embeddings, persist_directory="./chroma_db" ) def add_documents(chunks): vector_store.add_documents(chunks) vector_store.persist() def search_documents(query: str, k: int = 4) -> list: return vector_store.similarity_search(query, k=k)

3. RAG Pipeline — الإجابة على الأسئلة

# rag.py from openai import OpenAI from app.vector_store import search_documents from app.config import settings client = OpenAI(api_key=settings.openai_api_key) def ask_question(question: str) -> str: # 1. استرجاع المستندات ذات الصلة docs = search_documents(question, k=4) context = "\n\n".join([doc.page_content for doc in docs]) # 2. بناء الـ prompt prompt = f"""أنت مساعد ذكي. أجب السؤال بناءً على السياق التالي: السياق: {context} السؤال: {question} الإجابة:""" # 3. توليد الإجابة response = client.chat.completions.create( model="gpt-4", messages=[ {"role": "system", "content": "أنت خبير مطلع على المستندات المقدمة."}, {"role": "user", "content": prompt} ], temperature=0.3 ) return response.choices[0].message.content

4. FastAPI Routers

# routers/documents.py — رفع المستندات from fastapi import APIRouter, UploadFile, File from app.ingestion import load_and_chunk from app.vector_store import add_documents router = APIRouter(prefix="/documents", tags=["documents"]) @router.post("/upload") async def upload_document(file: UploadFile = File(...)): content = await file.read() # حفظ الملف مؤقتاً temp_path = f"/tmp/{file.filename}" with open(temp_path, "wb") as f: f.write(content) # معالجة وتخزين chunks = load_and_chunk(temp_path) add_documents(chunks) return {"message": f"تم معالجة {len(chunks)} مقطع", "filename": file.filename}
# routers/chat.py — الأسئلة from fastapi import APIRouter from pydantic import BaseModel from app.rag import ask_question router = APIRouter(prefix="/chat", tags=["chat"]) class QuestionRequest(BaseModel): question: str class AnswerResponse(BaseModel): answer: str sources: list[str] @router.post("/ask", response_model=AnswerResponse) def ask(req: QuestionRequest): # استرجاع + توليد answer = ask_question(req.question) # جلب المصادر (اختياري) from app.vector_store import search_documents docs = search_documents(req.question, k=2) sources = [doc.metadata.get("source", "") for doc in docs] return AnswerResponse(answer=answer, sources=sources)

5. Main — تجميع كل شيء

# main.py from fastapi import FastAPI from fastapi.middleware.cors import CORSMiddleware from app.routers import documents, chat app = FastAPI(title="AI Document Chat") app.add_middleware( CORSMiddleware, allow_origins=["*"], allow_methods=["*"], allow_headers=["*"], ) app.include_router(documents.router) app.include_router(chat.router) @app.get("/") def root(): return {"message": "مرحباً! ارفع مستنداتك وابدأ الأسئلة."}
# requirements.txt fastapi[standard] langchain langchain-community chromadb sentence-transformers pypdf openai uvicorn

6. التمارين

تمرين 1: تشغيل المشروع
ارفع ملف PDF (مثل كتاب أو مقال)، ثم اسأل أسئلة POST /chat/ask.
تمرين 2: History
أضف حفظ تاريخ المحادثة لكل جلسة session (باستخدام Redis).
تمرين 3: Chunk Size
جرب chunk_size مختلف (250, 500, 1000) ولاحظ الفرق في جودة الإجابات.
